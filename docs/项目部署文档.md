## 四、MySQL 8.0安装

### 4.1 下载并安装MySQL
```bash
cd /usr/local
wget https://dev.mysql.com/get/mysql80-community-release-el7-3.noarch.rpm
rpm -ivh mysql80-community-release-el7-3.noarch.rpm
yum install -y mysql-community-server
```

### 4.2 启动MySQL
```bash
systemctl start mysqld
systemctl enable mysqld
```

### 4.3 获取临时密码
```bash
grep 'temporary password' /var/log/mysqld.log
```

### 4.4 修改root密码
```bash
mysql -uroot -p

# 输入临时密码后执行
ALTER USER 'root'@'localhost' IDENTIFIED BY 'YourStrongPassword@123';
FLUSH PRIVILEGES;
```

### 4.5 创建数据库和用户
```sql
-- 创建数据库
CREATE DATABASE youyinda DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 创建用户
CREATE USER 'youyinda'@'%' IDENTIFIED BY 'YourDatabasePassword@123';

-- 授权
GRANT ALL PRIVILEGES ON youyinda.* TO 'youyinda'@'%';
FLUSH PRIVILEGES;
```

### 4.6 导入初始化脚本
```bash
mysql -uyouyinda -p youyinda < init.sql
mysql -uyouyinda -p youyinda < config_init.sql
```

### 4.7 MySQL配置优化
```bash
vim /etc/my.cnf

# 添加以下配置
[mysqld]
character-set-server=utf8mb4
collation-server=utf8mb4_unicode_ci
max_connections=500
innodb_buffer_pool_size=1G
innodb_log_file_size=256M
slow_query_log=1
slow_query_log_file=/var/log/mysql-slow.log
long_query_time=2
```

## 五、Redis 6.x安装

### 5.1 下载并安装Redis
```bash
cd /usr/local
wget https://download.redis.io/releases/redis-6.2.6.tar.gz
tar -zxvf redis-6.2.6.tar.gz
cd redis-6.2.6
make
make install
```

### 5.2 配置Redis
```bash
mkdir -p /etc/redis
cp redis.conf /etc/redis/6379.conf
vim /etc/redis/6379.conf

# 修改以下配置
daemonize yes
bind 0.0.0.0
requirepass YourRedisPassword@123
maxmemory 2gb
maxmemory-policy allkeys-lru
appendonly yes
```

### 5.3 启动Redis
```bash
# 创建服务脚本
vim /etc/systemd/system/redis.service

[Unit]
Description=redis-server
After=network.target

[Service]
Type=forking
ExecStart=/usr/local/bin/redis-server /etc/redis/6379.conf
PrivateTmp=true

[Install]
WantedBy=multi-user.target

# 启动服务
systemctl daemon-reload
systemctl start redis
systemctl enable redis
```

### 5.4 验证Redis
```bash
redis-cli -a YourRedisPassword@123
ping
# 返回 PONG
```

## 六、后端服务部署（Jar包方式）

### 6.1 编译打包
```bash
cd youyinda-backend
mvn clean package -DskipTests
```

### 6.2 上传Jar包
```bash
# 将target/youyinda-backend.jar上传到服务器
mkdir -p /opt/youyinda
cd /opt/youyinda
```

### 6.3 创建配置文件
```bash
vim application-prod.yml

server:
  port: 8080

spring:
  datasource:
    url: jdbc:mysql://localhost:3306/youyinda?useUnicode=true&characterEncoding=utf8&serverTimezone=Asia/Shanghai&useSSL=false
    username: youyinda
    password: YourDatabasePassword@123
    driver-class-name: com.mysql.cj.jdbc.Driver

  redis:
    host: localhost
    port: 6379
    password: YourRedisPassword@123
    database: 0

# 微信支付配置
wx:
  pay:
    app-id: your-app-id
    mch-id: your-mch-id
    api-key: your-api-key
    cert-path: /path/to/cert.p12
```

### 6.4 创建启动脚本
```bash
vim start.sh

#!/bin/bash
APP_NAME=youyinda-backend.jar
APP_HOME=/opt/youyinda
LOG_HOME=/opt/youyinda/logs

mkdir -p $LOG_HOME

nohup java -jar \
  -Xms512m \
  -Xmx1024m \
  -XX:+UseG1GC \
  -Dspring.profiles.active=prod \
  $APP_HOME/$APP_NAME \
  > $LOG_HOME/youyinda.log 2>&1 &

echo $! > $APP_HOME/app.pid
echo "服务启动成功，PID: $(cat $APP_HOME/app.pid)"
```

### 6.5 创建停止脚本
```bash
vim stop.sh

#!/bin/bash
APP_HOME=/opt/youyinda

if [ -f $APP_HOME/app.pid ]; then
  PID=$(cat $APP_HOME/app.pid)
  kill -15 $PID
  rm -f $APP_HOME/app.pid
  echo "服务已停止"
else
  echo "服务未运行"
fi
```

### 6.6 创建重启脚本
```bash
vim restart.sh

#!/bin/bash
./stop.sh
sleep 5
./start.sh
```

### 6.7 设置执行权限
```bash
chmod +x start.sh stop.sh restart.sh
```

### 6.8 启动服务
```bash
./start.sh

# 查看日志
tail -f logs/youyinda.log
```

## 七、Nginx安装与配置

### 7.1 安装Nginx
```bash
# CentOS
yum install -y nginx

# Ubuntu
apt-get install -y nginx
```

### 7.2 配置Nginx
```bash
vim /etc/nginx/conf.d/youyinda.conf

server {
    listen 80;
    server_name api.youyinda.com;

    access_log /var/log/nginx/youyinda-access.log;
    error_log /var/log/nginx/youyinda-error.log;

    client_max_body_size 50M;

    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/x-javascript application/xml+rss application/json application/javascript;

    location /api {
        proxy_pass http://localhost:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_connect_timeout 5s;
        proxy_send_timeout 30s;
        proxy_read_timeout 30s;
    }
}
```

### 7.3 启动Nginx
```bash
nginx -t
systemctl start nginx
systemctl enable nginx
```

## 八、Docker容器化部署

### 8.1 安装Docker
```bash
curl -fsSL https://get.docker.com | bash -s docker --mirror Aliyun
systemctl start docker
systemctl enable docker
```

### 8.2 安装Docker Compose
```bash
curl -L "https://github.com/docker/compose/releases/download/v2.20.0/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
chmod +x /usr/local/bin/docker-compose
```

### 8.3 创建Dockerfile
```dockerfile
FROM openjdk:8-jre-alpine

LABEL maintainer="youyinda"

WORKDIR /opt/youyinda

COPY target/youyinda-backend.jar app.jar

EXPOSE 8080

ENTRYPOINT ["java", "-jar", "-Xms512m", "-Xmx1024m", "-XX:+UseG1GC", "-Dspring.profiles.active=prod", "app.jar"]
```

### 8.4 创建docker-compose.yml
```yaml
version: '3.8'

services:
  mysql:
    image: mysql:8.0
    container_name: youyinda-mysql
    environment:
      MYSQL_ROOT_PASSWORD: YourRootPassword@123
      MYSQL_DATABASE: youyinda
      MYSQL_USER: youyinda
      MYSQL_PASSWORD: YourDatabasePassword@123
    ports:
      - "3306:3306"
    volumes:
      - ./mysql/data:/var/lib/mysql
      - ./mysql/conf:/etc/mysql/conf.d
      - ./init.sql:/docker-entrypoint-initdb.d/init.sql
    command:
      - --character-set-server=utf8mb4
      - --collation-server=utf8mb4_unicode_ci
    networks:
      - youyinda-network

  redis:
    image: redis:6.2-alpine
    container_name: youyinda-redis
    command: redis-server --requirepass YourRedisPassword@123 --appendonly yes
    ports:
      - "6379:6379"
    volumes:
      - ./redis/data:/data
    networks:
      - youyinda-network

  app:
    build: .
    container_name: youyinda-app
    ports:
      - "8080:8080"
    environment:
      SPRING_PROFILES_ACTIVE: prod
      SPRING_DATASOURCE_URL: jdbc:mysql://mysql:3306/youyinda?useUnicode=true&characterEncoding=utf8&serverTimezone=Asia/Shanghai&useSSL=false
      SPRING_DATASOURCE_USERNAME: youyinda
      SPRING_DATASOURCE_PASSWORD: YourDatabasePassword@123
      SPRING_REDIS_HOST: redis
      SPRING_REDIS_PORT: 6379
      SPRING_REDIS_PASSWORD: YourRedisPassword@123
    depends_on:
      - mysql
      - redis
    volumes:
      - ./logs:/opt/youyinda/logs
    networks:
      - youyinda-network
    restart: unless-stopped

  nginx:
    image: nginx:alpine
    container_name: youyinda-nginx
    ports:
      - "80:80"
    volumes:
      - ./nginx/conf.d:/etc/nginx/conf.d
      - ./nginx/logs:/var/log/nginx
    depends_on:
      - app
    networks:
      - youyinda-network
    restart: unless-stopped

networks:
  youyinda-network:
    driver: bridge
```

### 8.5 启动Docker容器
```bash
docker-compose up -d

# 查看日志
docker-compose logs -f app
```

## 九、小程序发布流程

### 9.1 代码上传
1. 打开微信开发者工具
2. 选择「上传」按钮
3. 填写版本号和项目备注
4. 点击「确定」上传代码

### 9.2 提交审核
1. 登录[微信公众平台](https://mp.weixin.qq.com/)
2. 进入「版本管理」
3. 找到刚上传的开发版本
4. 点击「提交审核」
5. 填写审核信息：
   - 功能页面：首页、打印、快递、订单、我的
   - 功能描述：云打印和快递寄件服务
6. 提交审核

### 9.3 发布上线
1. 等待审核通过（一般1-7个工作日）
2. 审核通过后，点击「发布」
3. 确认发布信息
4. 完成发布，用户即可搜索到小程序

### 9.4 灰度发布（可选）
- 支持按比例灰度发布
- 先发布给10%用户
- 观察24小时无问题后全量发布

## 十一、安全加固方案

### 11.1 系统安全
1. **定期更新系统补丁**
```bash
yum update -y
```

2. **禁用root远程登录**
```bash
vim /etc/ssh/sshd_config
PermitRootLogin no
```

3. **修改SSH端口**
```bash
vim /etc/ssh/sshd_config
Port 22222
```

4. **使用密钥登录**
```bash
ssh-keygen -t rsa
ssh-copy-id -i ~/.ssh/id_rsa.pub user@server
```

### 11.2 数据库安全
1. **删除匿名用户**
```sql
DELETE FROM mysql.user WHERE User='';
```

2. **禁止root远程登录**
```sql
DELETE FROM mysql.user WHERE User='root' AND Host NOT IN ('localhost', '127.0.0.1');
```

3. **定期备份**
```bash
# 创建备份脚本
vim /opt/backup/mysql-backup.sh

#!/bin/bash
DATE=$(date +%Y%m%d_%H%M%S)
mysqldump -uyouyinda -p'YourPassword' youyinda | gzip > /opt/backup/youyinda_$DATE.sql.gz
find /opt/backup -name "youyinda_*.sql.gz" -mtime +7 -delete
```

### 11.3 应用安全
1. **HTTPS配置**
```nginx
server {
    listen 443 ssl;
    server_name api.youyinda.com;

    ssl_certificate /path/to/cert.pem;
    ssl_certificate_key /path/to/key.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;

    location /api {
        proxy_pass http://localhost:8080;
    }
}

server {
    listen 80;
    server_name api.youyinda.com;
    return 301 https://$server_name$request_uri;
}
```

2. **接口限流**
- 使用Spring Security + Redis实现限流
- 单用户每秒最多10次请求
- 单IP每秒最多100次请求

3. **敏感信息加密**
- 用户手机号、身份证号等敏感信息AES加密存储
- 微信支付密钥等配置使用Jasypt加密

## 十三、回滚方案

### 13.1 后端回滚
```bash
# 停止当前版本
./stop.sh

# 备份当前版本
cp youyinda-backend.jar youyinda-backend.jar.bak

# 回滚到上一版本
cp youyinda-backend.jar.v1.0.0 youyinda-backend.jar

# 启动
./start.sh
```

### 13.2 数据库回滚
```bash
# 使用备份文件恢复
gunzip < /opt/backup/youyinda_20240101_120000.sql.gz | mysql -uyouyinda -p youyinda
```

### 13.3 小程序回滚
- 微信小程序支持版本回退
- 在版本管理中选择「回退」到上一版本
